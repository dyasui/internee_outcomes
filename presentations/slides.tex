\begin{frame}{Introduction}
  This project asks:
  \begin{itemize}
    \item How did Japanese American internment during WWII 
    affect long-run labor market outcomes?
    \item Can forced relocation act as a “move-to-opportunity” shock?
    \item Do certain types of people adapt differently?
  \end{itemize}
  I find that:
  \begin{itemize}
    \item Internment paradoxically lead to higher incomes in 1950
    \item Internees moved to higher-income counties
    \item Interned women had higher income growth than men 
  \end{itemize}
\end{frame}

\begin{frame}
  \begin{center}
    \textbf{Why should we care?}
  \end{center}
  \begin{figure}
    \includegraphics[width=0.8\textwidth]{UNHCR_forced_displacement_2024.png}\\
    {\tiny Source: \href{https://www.unhcr.org/global-trends-report-2024}{UNHCR global trends report}}
  \end{figure}
  Understanding effects of forced migration is important
  \begin{itemize}
    \item Forced migrants can adapt by finding new opportunities \footcite{bauerEconomicIntegrationForced2013,nakamuraGiftMovingIntergenerational2022}
    \item Migration frictions might keep people from pursuing labor market opportunities \footcite{munshiSocialNetworksMigration2020,abramitzkyLeavingEnclaveHistorical2024,deryuginaDoesWhenYou2020}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{What can Japanese Internment teach us?}
  Internment displaced $>$100,000 Japanese Americans from 1942-1945
  \begin{itemize}
    \item Internees lost property, employment opportunities,
    experienced worse school quality, and childhood mortality during incarceration
    \footcite{chinLongRunLaborMarket2005,saavedraEarlyChildhoodConditions2013,saavedraSchoolQualityEducational2015}
    \item However, emerging empirical literature points to 
    unexpected upward mobility in the long-run
      \begin{itemize}
        \item Internees were exposed to new occupation and migration possibilities \footcite{arellano-boverDisplacementDiversityMobility2022}
        \item Camp assignment led to quasi-random long-term resettlement \footcite{shoagCausalEffectPlace2016}
      \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  What I contribute:
  \begin{itemize}
    \item Use new individual-level linked Census data (1940-1950)
    \item Causal evidence on the migration mechanism leading to internment income effect
    \item New evidence on heterogenous outcomes by gender
  \end{itemize}
\end{frame}

\begin{frame}{Predicted Internment Status}\label{intstatus}
  The intuition:
  \begin{itemize}
    \item Most mainland Japanese were interned, but not all
    \item Chinese Americans were never interned, but had similar migrant experience
    \item Non-interned Japanese were either already living outside of Exclusion Area,
    or managed to leave between 1940 and 1942.
    \item Voluntary evacuation was likely endogenous:
    \begin{itemize}
      \item some Japanese students transfered to non-West Coast colleges
      \item some had families or professional connections in other states
    \end{itemize}
  \end{itemize}
  The solution:
    \begin{itemize}
      \item Compare county and demographic records in WRA camps to linked sample
      \item Update naive prior that all West Coast Japanese were interned with more detailed information
      \item Probability of internment is higher for more affected counties, always zero for Chinese
    \end{itemize}
  \hyperlink{bayesrule}{\beamergotobutton{Details}}
\end{frame}

\begin{frame}{Econometric Specification}
  Panel of Japanese and Chinese individuals born before 1926 with employment data in 1940 and 1950
  \begin{equation*}\label{eq:mainreg}
    \text{Income}_{i,1950} = \alpha + \beta ~ \widehat{\text{Internment}} 
    + \mathbb{X}_i \Gamma + \delta \text{Income}_{i,1940} +
    \epsilon_i
  \end{equation*}
  \begin{itemize}
    \item individual $i$ is observed in both 1940 and 1950
    \item Income from wages across both census years (in 1950 dollars)
    \item $\mathbb{X}_i$ time-invariant demographics: age, gender, race
  \end{itemize}
\end{frame}

\begin{frame}{Methods}\label{methods}
  \begin{block}{Difference-in-Differences}
    Assuming that Chinese and Japanese Americans would have experienced similar wage trends without internment:
    \begin{align*}
    \text{Income}_{i,t} & = \beta_0 \mathbb{I}(t=1950) + \beta_1 ~ \widehat{\text{Internment}} \\
                        & + \beta_{DiD} \widehat{\text{Internment}} \times \mathbb{I}(t=1950) \\
                        & + \mathbb{X}_i \Gamma + \epsilon_{it}
    \end{align*}
  \end{block}
    \begin{itemize}
    \item Income$_{it}$: real wages earned by individual $i$ in year $t\in\{1940,1950\}$
    \item $\mathbb{I}$: treatment has an effect in 1950, but not 1940
    \item $X_{it}$: Time-variant controls (age and race)
    \end{itemize}
    $\beta_{DiD}$ estimate between internees and non-internees, pre vs post internment
\end{frame}

\begin{frame}{Mechanisms}
  One possible explanation:
  \begin{itemize}
    \item Internees responded to lost wages by increasing labor market participation
    \item Would expect to see measures of intensive and extensive participation increase
    \item Also might expect to see shifting to higher-earning occupations
    \item Census asks employment, hours and weeks worked
    \item Occupational categories ranked by median earning within same category
  \end{itemize}
  This doesn't seem to be the case, if anything internees switched jobs less frequently
\end{frame}

\begin{frame}
  \frametitle{Mechanisms}
  Another explanation is that internment kicked off `moves to opportunity'
  \begin{itemize}
    \item After release from camps, many moved to new cities like Denver, Chicago, New York
    \item New geographic access to labor markets with more opportunities
  \end{itemize}
  The evidence seems more consistent with this mechanism:
  \begin{itemize}
    \item Hard to define migration status for internees who move back to West Coast
    \item But internees who move to new counties end up in higher-income areas their migrant peers
  \end{itemize}
\end{frame}

\begin{frame}{Discussion}
  Takeaways:
  \begin{itemize}
    \item Internment produced a forced Move to Opportunity shock for many internees.
    \item Migration, not occupation switching, is the key channel.
    \item Gender norms \& labor market structures likely shaped heterogeneous outcomes.
  \end{itemize}
  Caveats:
  \begin{itemize}
    \item Does not measure:
    \item wealth loss
    \item trauma
    \item property seizure
  \end{itemize}
  Results do not diminish moral harm of internment.
\end{frame}

\begin{frame}{Methods}\label{bayesrule}
  \addtocounter{framenumber}{-1}
  \begin{block}{Predicted Internment Status}
    Using known proportions of demographic categories within WRA and census,
    we can estimate the probability of internment using Bayes Rule:
    $$
    Pr(I_i=1|z_i, c(i)) = 
    \frac{Pr\left(c(i), z_i|I_i=1\right) \cdot Pr(I_i=1)}{Pr(c(i), z_i)}.
    $$
    \begin{itemize}
    \item $Pr(z_i,c(i)|I_i=1)$: proportion of internees in each county with demographics $z$
    \item $Pr(I_i=1)$: fraction of internees over 1940 Japanese American population
    \item $Pr(z_i,c(i))$: proportion of Japanese Americans in each county-demographic in 1940 census
    \end{itemize}
  \end{block}
  \hyperlink{intstatus}{\beamergotobutton{Back}}
\end{frame}
