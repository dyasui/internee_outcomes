collect_cb <- function(x, pos) {
  x |>
    filter(RACE %in% c(4,5),
           !(INCWAGE %in% c(999999, 999998, NA) ) ) |>
    mutate(INCWAGE = ifelse(YEAR == 1940, INCWAGE * 1.71, INCWAGE) ) |>
    select(YEAR, STATEFIP, COUNTYICP, SEX, AGE, BIRTHYR, BPL, RACE, EDUC, LABFORCE, OCC1950, EMPSTAT, INCWAGE)
}

fc_data <- read_ipums_micro_chunked(
  ddi,
  callback = IpumsDataFrameCallback$new(collect_cb),
  chunk_size = 1e5,
  verbose = TRUE
)
#+end_src

** NOTE this is preliminary until I figure out how to estimate internees migrations from 1940-1950

This is using respondent's county in 1950 as a proxy for their 1940 location to preduct their internment status.
This assumes (incorrectly) that no migrations of internees occured between 1940 and 1950,
but it gives an idea of the lasting effect of internment on the county-level Japanese community as well as the effect of internment on the internees who chose to move back.
#+begin_src R
fcreg_df <- predict_internment(fc_data, vars = c("STATEFIP", "COUNTYICP", "RACE", "SEX" )) |>
  mutate(
    i_1950 = ifelse(YEAR==1950, 1, 0),
    west_coast = case_when(
      # all of california evacuation zone
      STATEFIP==6 ~ 1,
      # Arizona evacuation zones
      (STATEFIP==4 & COUNTYICP %in% c(30, 130, 190, 230, 270)) ~ 1,
      # Oregon evacuation zones
      (STATEFIP==41 & COUNTYICP %in% c(30, 50, 70, 90, 110, 190, 270, 290, 330, 390, 410, 430, 470, 510, 530, 570, 670, 690, 730, 770)) ~ 1,

      .default = 0
