sampleline_sample <- linked_data_full |>
  mutate(sampleline_1950 = case_when(
           YEAR==1950 & SLWT!=0 ~ 1,
           YEAR==1950 & SLWT==0 ~ 0,
           YEAR==1940 ~ 1
         )) |> 
  filter(birthyr < 1925, LABFORCE==2, sampleline_1950==1) |>
  mutate(pr_intern = ifelse(RACE==4, 0, prI_county),
         pr_intern2 = ifelse(RACE==4, 0, prI_full),
         pr_intern3 = ifelse(RACE==4, 0, prI_AB),
         across(c("STATEFIP", "BPL", "OCC1950", "MIGRATE5"), as_factor)
         ) |>
  group_by(id) |>
  filter(n() >= 2) |>
  ungroup() |>
  pivot_wider(
    id_cols = c("id", "race", "sex", "pr_intern", "pr_intern2", "pr_intern3", "migrate10", "income_tot"),
    names_from = YEAR,
    values_from = c("age", "salary", "employed", "OCC1950",
                    "OCCSCORE", "WKSWORK1", "yearsschool", "STATEFIP",
                    "county_med.salary", "county_pop", "county_pop.japanese")
  )
#+end_src
