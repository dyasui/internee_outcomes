\section{Research Question}
%---------------------------------------------------------------------------

\begin{frame}{Research Question}\label{rq_slide}
    \begin{block}{Research Question:}
        \textbf{How were internee labor market outcomes affected after their release?}
    \end{block}
    \begin{itemize}
        \item WRA camps and resettlement lead to shift in Japanese Population \hyperlink{areaplot}{\beamergotobutton{Plot}}
        \item Variation in camp areas by local median incomes \hyperlink{stincomemap}{\beamergotobutton{Map}}
        \item Camps assigned by geography and capacity; not economic characteristics \hyperlink{campplcmap}{\beamergotobutton{Map}}
    \end{itemize}
\end{frame}

%---------------------------------------------------------------------------
\section{Significance and Contribution}
%---------------------------------------------------------------------------

\begin{frame}{Related Literature}
    \begin{block}{Effects on Japanese American Internees:}
        \begin{itemize}
            \item Internment caused labor market withdrawl leading to reduction of working-age male earnings by 9-13\% in 1970 \citep{chin_longrun_2005}
            \item Internment may have lead to higher long-term wages due to re-optimization and camp diversity (Arellano-Bover 2022).
            \item Camp assignment induced resettlement and those moved to richer areas did better then those moved to poorer areas in 1980s \citep{shoag_causal_2016}
        \end{itemize}
    \end{block}
\end{frame}

    \begin{frame}{Methods}
      \begin{block}{Predicted Internment Status}
        Using known proportions of demographic categories within WRA and census,
        we can estimate the probability of internment using Bayes Rule:
        $$
        Pr(I_i=1|Z_i,c_i^{40}) = \frac{Pr(z_i,c_i^{40}|I_i=1) \cdot Pr(I_i=1)}{Pr(z_i,c_i^{40})}
        $$
        \begin{itemize}
        \item $Pr(z_i,c_i^{40}|I_i=1)$: proportion of internees in each county-demographic
        \item $Pr(I_i=1)$: fraction of internees over 1940 Japanese American population
        \item $Pr(z_i,s_i^{40})$: proportion of Japanese Americans in each county-demographic in 1940 census
        \end{itemize}
      \end{block}
    \end{frame}

    \begin{frame}{Methods}\label{methods}
      \begin{block}{Difference-in-Differences \hyperlink{DiDmethod}{\beamergotobutton{Diff-in-Diff}}}
        Assuming that Chinese and Japanese Americans would have experienced similar wage trends without internment:
        $$y_{it} = \alpha_t + X_{it}' \gamma + \delta \hat{E}[I_i|Z_i,c_i^t] + \beta (\hat{E}[I_i|Z_i,c_i^t] \times Post_t) + \epsilon_{it}$$
      \end{block}
        \begin{itemize}
        \item $y_{it}$: real wages earned by individual $i$ in year $t$
        \item $\hat{E}[I_i|Z_i,c_i^t]$: Predicted internhment status
        \item $Post_t$: treatment has an effect in 1950, but not 1940
        \item $X_{it}$: Time-variant controls (educ, marst, etc)
        \end{itemize}
        $\beta$ : DiD estimate between internees and non-internees, pre vs post internment
    \end{frame}

    \begin{frame}{Diff-in-Diff}
      \begin{block}{Identifying assumption:}
        The only reason an internee would resettle and stay in destination $k$ is through the internment and resettlement mechanism
      \end{block} 
      \begin{itemize}
      \item Possibly violated if internees did have some degree of choice and ability to move between release and 1950 census
      \end{itemize}
    \end{frame}
