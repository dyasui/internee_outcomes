# Collect average wages by racial group from database
callback <- function(x, pos) {
  x |>
    group_by(YEAR, RACE) |>
    mutate(
      salary = case_when(
        INCWAGE %in% c(999999, 999998) ~ NA,
        YEAR == 1940 ~ INCWAGE * 1.69,
        YEAR == 1950 ~ INCWAGE
      )
    ) |>
    filter(!is.na(salary)) |>
    summarise(
      k = n(),
      mean = mean(salary, na.rm = T),
      var = var(salary, na.rm = T)
      )
}

combine_var <- function(stat1, stat2) {
  n1 <- stat1$n; x1 <- stat1$mean; sx1 <- stat1$var
  n2 <- stat2$n; x2 <- stat2$mean; sx2 <- stat2$var
  
  n_combined <- n1 + n2
  mean_combined <- (n1 * x1 + n2 * x2) / n_combined
  var_combined <- ((n1 - 1) * sx1 + (n2 - 1) * sx2) / (n_combined - 1) + 
                  n1 * n2 * (x1 - x2)^2 / (n_combined * (n_combined - 1))
  
  list(n = n_combined, mean = mean_combined, var = var_combined)
}

# apply combine_var over subsample sum stats
total_var <- function(chunks, n, mean, var, groups) {
  library(purrr)
  cumulative_stats <- chunks |>
    # if group only has one observation, drop it
    filter({{n}} > 1) |>
    group_by({{ groups }}) |>
    mutate(
      stats = map(row_number(),
                  ~ list(n = {{n}}[.x],
                         mean = {{mean}}[.x],
                         var = {{var}}[.x])
                  )
      ) |>
    pull(stats) |>
    reduce(combine_var,
               .init = list(
                 n = 0,
                 mean = 0,
                 var = 0)) # |>
    ## tail(-1) |>
    ## map_dfr(~ tibble(n = .x$n, mean = .x$mean, var = .x$var),
    ##         .id = "cumulative_step")

  cumulative_stats
}

plt_data <- group_salaries |>
  group_by(YEAR, RACE) |>
  # map total variance formula to each group's tibble
  group_modify(~ {
    result <- total_var(chunks = .x, n = k, mean = mean, var = var)
    tibble(
      total_n = result$n,
      mean = result$mean, 
      var = result$var
    )
  }) |>
  mutate(
    sd = sqrt(var),
    RACE = case_when(
      RACE==1 ~ "White",

      RACE==3 ~ "Native",
      RACE==4 ~ "Chinese",
      RACE==5 ~ "Japanese",
      RACE==6 ~ "Asian other",
      RACE==7 ~ "Other"
    )
  )
