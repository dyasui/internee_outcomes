theme_internment <- function(){ 
    font <- "Georgia"   #assign font family up front
    
    theme_bw() %+replace%    #replace elements we want to change
    
    theme(
      
      #grid elements
      panel.grid.major = element_blank(),    #strip major gridlines
      panel.grid.minor = element_blank(),    #strip minor gridlines
      axis.ticks = element_blank(),          #strip axis ticks
      
      #text elements
      plot.title = element_text(             #title
                   family = font,            #set font family
                   size = 14,                #set font size
                   face = 'bold',            #bold typeface
                   hjust = 0,                #left align
                   vjust = 2),               #raise slightly
      
      plot.subtitle = element_text(          #subtitle
                   family = font,            #font family
                   size = 14),               #font size
      
      ## plot.caption = element_text(           #caption
      ##              family = font,            #font family
      ##              size = 9,                 #font size
      ##              hjust = 1),               #right align
      
      axis.title = element_text(             #axis titles
                   family = font,            #font family
                   size = 10),               #font size
      
      axis.text = element_text(              #axis text
                   family = font,            #axis famuly
                   size = 9),                #font size
      
      axis.text.x = element_text(            #margin for axis text
                    margin=margin(5, b = 10))
      
    )
}

library(sf)

census_dat <- read_csv("data/county_stats.csv") |>
  filter(YEAR==1940) |> 
  mutate(
    NHGISST = str_pad(string = STATEFIP*10, width = 3, side = "left", pad = "0"),
    NHGISCTY = str_pad(COUNTYICP, width = 4, side = "left", pad = "0")
  )
int_dat <- read_csv("data/county_internment.csv")

map_dat <- left_join(census_dat, int_dat,
                     by = c("NHGISST", "NHGISCTY")) |>
  mutate(num_int = ifelse(is.na(n), 0, n))

# read shapefiles and append distances
shp <- ipumsr::read_ipums_sf("data/nhgis0035_shape.zip", file_select = "nhgis0035_shape/nhgis0035_shapefile_tl2008_us_county_1950.zip") |>
  filter( STATENAM %in% c("Oregon", "Washington", "California", "Arizona") ) |>
  left_join(map_dat, by = c("NHGISST", "NHGISCTY")) |>
  mutate(pr_interned = num_int / pop_jp) |>
  mutate(pr_interned = ifelse(pr_interned > 1, 1, pr_interned))

pr_interned_map <- ggplot(data = shp, aes(geometry=geometry)) +
  geom_sf(aes(fill=pr_interned))
