#+begin_src R
# Collect 1940 fullcount census demography and location data from IPUMS download
ddi <- "data/fullcount_census/usa_00128.xml"

# callback function to calculate county summary stats
jpstat_cb <- function(x, pos) {
  x |>
    filter(RACE==5, YEAR==1940) |>
    mutate(across(c("CITY", "OCC1950"), as_factor),
           BIRTHYR = ifelse(BIRTHYR %in% 9996:9999 , NA, BIRTHYR)
           )
}

jp_demo <- read_ipums_micro_chunked(
  ddi,
  vars = c("YEAR", "RACE", "SEX", "BIRTHYR", "BPL", "EDUC", "EDUCD", "OCC1950", "STATEFIP", "COUNTYICP", "CITY", "FBPL", "MBPL", "NATIVITY"),
  callback = IpumsDataFrameCallback$new(jpstat_cb),
  chunk_size = 1e5,
  verbose = TRUE
) |>
  bind_rows()
#+end_src

#+begin_src R :results silent
# merge with internee data to compare
cen_demo <- jp_demo |>
  mutate(
    NHGISST = str_pad(string = STATEFIP*10, width = 3, side = "left", pad = "0"),
    NHGISCTY = str_pad(COUNTYICP, width = 4, side = "left", pad = "0"),
    source = "1940 Census",
  ) |>
  left_join(read_csv("data/nhgis_codes.csv"), by = c("NHGISST", "NHGISCTY")) |> 
  rename(state="STATENAM", county="NHGISNAM") |> 
  select(source, SEX, BIRTHYR, BPL, state, county)

int_demo <- wra_data |>
  filter(race==5) |> 
  mutate(
    source = "Internees",
  ) |>
  rename(SEX="sex",BIRTHYR="birthyr",BPL="bpl") |> 
  select(source, SEX, BIRTHYR, BPL, state, county)

tbl_dt <- bind_rows(cen_demo, int_demo) |>
  mutate(
    birth_cntry = case_when(
      BPL %in% 1:99 ~ "USA",
      BPL == 501 ~ "Japan",
      .default = "Other"
    )
  )
#+end_src

#+begin_src R :exports results :results graphics file :file figures/compare-age_plot.pdf
